Uses the standard openai npm package — no custom SDK needed. Just change baseURL.

typescript 
import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.CUSTOM_ROUTER_API_KEY,
  baseURL: "https://your-router.example.com/api/v1",
});

// Let the router pick the best model automatically
const res = await client.chat.completions.create({
  model: "auto",
  messages: [{ role: "user", content: "Write a hello world in Python." }],
});

// Or target a specific profile
const cheap = await client.chat.completions.create({
  model: "auto-cheap",
  messages: [{ role: "user", content: "Summarize this text..." }],
});

Uses the standard openai Python package. Works with LangChain, LlamaIndex, and any OpenAI-compatible library.

python 
from openai import OpenAI
import os

client = OpenAI(
    api_key=os.environ["CUSTOM_ROUTER_API_KEY"],
    base_url="https://your-router.example.com/api/v1",
)

# Smart routing — router selects the best model
response = client.chat.completions.create(
    model="auto",
    messages=[{"role": "user", "content": "Write hello world in Python."}],
)

# Or use a specific gateway model directly
response = client.chat.completions.create(
    model="openai/gpt-4o",
    messages=[{"role": "user", "content": "Explain quantum entanglement."}],
)

Use the Vercel AI SDK's createOpenAI provider with a custom baseURL. Works with streamText, generateText, and all other AI SDK helpers.

typescript 
import { createOpenAI } from "@ai-sdk/openai";
import { streamText } from "ai";

const router = createOpenAI({
  apiKey: process.env.CUSTOM_ROUTER_API_KEY,
  baseURL: "https://your-router.example.com/api/v1",
});

// Use smart routing in a Next.js Server Action / Route Handler
const result = await streamText({
  model: router("auto"),
  prompt: "Write a haiku about distributed systems.",
});

return result.toDataStreamResponse();

typescript — app/api/chat/route.ts 
import { createOpenAI } from "@ai-sdk/openai";
import { streamText } from "ai";

const router = createOpenAI({
  apiKey: process.env.CUSTOM_ROUTER_API_KEY,
  baseURL: "https://your-router.example.com/api/v1",
});

export async function POST(req: Request) {
  const { messages } = await req.json();

  const result = await streamText({
    model: router("auto"),
    messages,
  });

  return result.toDataStreamResponse();
}
